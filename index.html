const SQUARES = new Set([1,4,9,16,25,36,49,64,81,100,121,144]);

const COLS4 = ["#FF6B6B","#4ECDC4","#45B7D1","#FFEAA7"];

function genQ(mode) {

  const type = mode==="multiply"?"mult":mode==="divide"?"div":Math.random()<0.5?"mult":"div";

  const a=Math.floor(Math.random()*12)+1, b=Math.floor(Math.random()*12)+1;

  if(type==="mult"){const ans=a*b;return{text:a+" × "+b,answer:ans,isSquare:a===b&&SQUARES.has(ans)};}

  return{text:(a*b)+" ÷ "+a,answer:b,isSquare:false};

}

function genChoices(answer){

  const s=new Set([answer]);

  while(s.size<4){const c=answer+Math.floor(Math.random()*10)-5;if(c>0&&c!==answer)s.add(c);}

  return[...s].sort(()=>Math.random()-0.5);

}

function makeQ(mode){const q=genQ(mode);return{...q,choices:genChoices(q.answer)};}

function loadStorage(k){try{return JSON.parse(localStorage.getItem(k))||[];}catch(e){return[];}}

function saveStorage(k,v){try{localStorage.setItem(k,JSON.stringify(v));}catch(e){}}


function createPanelState(mode){

  return{q:makeQ(mode),score:0,stars:0,flash:null,showStar:false,answered:false,total:0,correct:0};

}

function panelReducer(state,action){

  if(action.type==="CORRECT")return{...state,answered:true,flash:"correct",score:state.score+1,total:state.total+1,correct:state.correct+1};

  if(action.type==="CORRECT_STAR")return{...state,answered:true,flash:"correct",score:state.score+2,stars:state.stars+1,showStar:true,total:state.total+1,correct:state.correct+1};

  if(action.type==="WRONG")return{...state,answered:true,flash:"wrong",total:state.total+1};

  if(action.type==="HIDE_STAR")return{...state,showStar:false};

  if(action.type==="NEXT")return{...state,q:makeQ(action.mode),answered:false,flash:null};

  return state;

}


function drawStar(ctx,cx,cy,spikes,outer,inner){

  let rot=(Math.PI/2)*3;const step=Math.PI/spikes;

  ctx.beginPath();ctx.moveTo(cx,cy-outer);

  for(let i=0;i<spikes;i++){ctx.lineTo(cx+Math.cos(rot)*outer,cy+Math.sin(rot)*outer);rot+=step;ctx.lineTo(cx+Math.cos(rot)*inner,cy+Math.sin(rot)*inner);rot+=step;}

  ctx.lineTo(cx,cy-outer);ctx.closePath();ctx.fill();

}

    function burst(x,y,count,speed){

      for(let i=0;i<count;i++){

          </div>

  useEffect(()=>{if(stars!==prevStars.current){prevStars.current=stars;onStarsChange(stars);}});

  useEffect(()=>{if(time===0&&!victory)setVictory(true);},[time]);

  const sched=(d)=>{nxt.current=setTimeout(()=>dispatch({type:"NEXT",mode}),d);};

  const pick=(c)=>{

    if(answered||time===0)return;

    if(c===q.answer){if(q.isSquare){dispatch({type:"CORRECT_STAR"});setTimeout(()=>{dispatch({type:"HIDE_STAR"});sched(150);},900);}else{dispatch({type:"CORRECT"});sched(380);}}

    else{dispatch({type:"WRONG"});sched(480);}

  };

  const tc=time>20?"#4ECDC4":time>10?"#f0a500":"#FF6B6B";

  const bg=flash==="correct"?"#d4f7d4":flash==="wrong"?"#ffd5d5":"#fff";

  const acc=total>0?Math.round((correct/total)*100):null;

  return(

    <div style={S.center}>

      {victory&&<VictoryScreen winnerName={username} isDraw={false} onDone={()=>{setVictory(false);onDone(score,stars,{total,correct});}}/>}

      <div style={{...S.card,position:"relative",background:bg,transition:"background 0.25s"}}>

        {showStar&&<StarBurst/>}

        <div style={{display:"flex",justifyContent:"space-between",alignItems:"center",marginBottom:10}}>

          <div style={{fontSize:"0.95rem"}}>⭐ {stars}</div>

          <div style={{fontSize:"2rem",fontWeight:"bold",color:tc}}>{time}s</div>

          <div style={{fontSize:"0.95rem"}}>Score: {score}</div>

        </div>

        {acc!==null&&<div style={{fontSize:"0.85rem",color:"#777",marginBottom:6}}>🎯 {acc}%</div>}

        <div style={{height:7,background:"#eee",borderRadius:4,marginBottom:14}}><div style={{height:"100%",background:tc,borderRadius:4,width:((time/60)*100)+"%",transition:"width 1s linear"}}/></div>

        {q.isSquare&&<div style={{background:"#fff8e1",border:"2px solid #ffd700",borderRadius:8,padding:"3px 10px",fontSize:"0.78rem",fontWeight:"bold",color:"#b8860b",marginBottom:8}}>⭐ Squared! +2 pts!</div>}

        <div style={{fontSize:"2rem",fontWeight:"bold",color:"#333",padding:"10px 0",textAlign:"center"}}>{q.text} = ?</div>

        <div style={{display:"grid",gridTemplateColumns:"1fr 1fr",gap:12,marginTop:12}}>

          {q.choices.map((c,i)=>{let cbg=COLS4[i];if(answered)cbg=c===q.answer?"#4CAF50":"#ccc";return<button key={i} onPointerDown={()=>pick(c)} style={{...S.choiceBtn,background:cbg,color:answered&&c!==q.answer?"#aaa":"#fff",touchAction:"manipulation",userSelect:"none"}}>{c}</button>;})}

        </div>

      </div>

    </div>

  );

}

  const speed=seconds>0?(seconds/total).toFixed(1):null;

  return(

    <div style={{display:"flex",gap:6,flexWrap:"wrap",justifyContent:"center",margin:"4px 0"}}>

      <div style={{background:"#f0f0f0",borderRadius:10,padding:"4px 12px",fontSize:"0.9rem",color:"#555"}}>
